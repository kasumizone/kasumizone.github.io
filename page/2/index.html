Element UI 后台管理系统主要的标签：

  • el-container：构建整个页面框架。
  • el-aside：构建左侧菜单。
  • el-menu：左侧菜单内容，常⽤属性：
    • :default-openeds：默认展开的菜单，通过菜单的 index 值来关联。
    • :default-active：默认选中的菜单，通过菜单的 index 值来关联。
  • el-submenu：可展开的菜单，常⽤属性：
    • index：菜单的下标，⽂本类型，不能是数值类型。
  • template：对应 el-submenu 的菜单名。
  • i：设置菜单图标，通过 class 属性实则。
    • el-icon-messae
    • el-icon-menu
    • el-icon-setting
  • el-menu-item：菜单的⼦节点，不可再展开，常⽤属性：
    • index：菜单的下标，⽂本类型，不能是数值类型。

ComicService的实现类ComicServiceImpl中

把ComicTypeVO转成comic才能存入数据库，那么跟上次渲染表单相反转换就好

@Override
public int saveData(ComicTypeVO comicTypeVO) {

    Comic comic = new Comic();
    BeanUtils.copyProperties(comicTypeVO, comic);            //comicTypeVO赋值给comic

    QueryWrapper<ComicType> wrapper = new QueryWrapper<>();

    wrapper.eq("name", comicTypeVO.getTypeone());
    comic.setTypeoneid(comicTypeMapper.selectOne(wrapper).getId());

    wrapper = new QueryWrapper<>();                            //注意每次都要new一个新的wrapper
    wrapper.eq("name", comicTypeVO.getTypetwo());
    comic.setTypetwoid(comicTypeMapper.selectOne(wrapper).getId());

    wrapper = new QueryWrapper<>();
    wrapper.eq("name", comicTypeVO.getTypethree());
    comic.setTypethreeid(comicTypeMapper.selectOne(wrapper).getId());

    //表单做了非空验证，所以这里可以不做非空判断，如果表单传来的数据有控制，会造成空指针异常
    int res = comicMapper.insert(comic);
    return res;
}

一、Shiro核心组件

用户，角色，权限

给角色赋予权限，给用户赋予角色

  1. UsernamePasswordToken，shiro用来封装用户登录信息，使用用户的登录信息来创建令牌 Token 。
  2. SecurityManager，Shiro核心部分，负责安全认证和授权。
  3. Subject，Shiro的一个抽象概念，包含了用户信息。
  4. realm，开发者自定义的模块，根据项目的需求，验证和授权的逻辑全部写在 Realm 中。
  5. AuthenticationInfo，用户的角色信息集合，认证时使用。
  6. AuthorizationInfo，角色的权限信息集合，授权时使用
  7. DefaultWebSecurityManager，安全管理器，开发者自定义的Realm需要注入到 DefaultWebSecurityManager 进行管理才能生效。
  8. ShiroFilterFactoryBean，过滤器工厂，Shiro的基本运行机制是开发者定制规则，Shiro去执行，具体的执行操作就是由 ShiroFilterFactoryBean 创建的一个个Filter对象来完成的。

package lover.alice.config;

import lover.alice.realm.AccountRealm;
import org.apache.shiro.spring.web.ShiroFilterFactoryBean;
import org.apache.shiro.web.mgt.DefaultWebSecurityManager;
import org.springframework.beans.factory.annotation.Qualifier;
import org.springframework.context.annotation.Bean;
import org.springframework.context.annotation.Configuration;

@Configuration
public class ShiroConfig {

    /**
     * 创建一个AccountRealm对象并且通过 @Bean 把他放到IOC容器中
     * 因为开发者自定义的Realm需要注入到 DefaultWebSecurityManager 进行管理才能生效。
     * 所以下一步
     */
    @Bean
    public AccountRealm accountRealm(){
        return new AccountRealm();
    }

    /**
     * @param accountRealm  上一步注册到IOC中的对象，现在通过 @Qualifier 取出来并且注入到DefaultWebSecurityManager中
     * @Qualifier 通过对象的名字，从容器中取出需要的对象，名字默认是方法名
     * 同样要把它注入到IOC容器中，供最后干事的 ShiroFilterFactoryBean 使用
     * 所以下一步
     */
    @Bean
    public DefaultWebSecurityManager securityManager(@Qualifier("accountRealm") AccountRealm accountRealm){
        DefaultWebSecurityManager manager = new DefaultWebSecurityManager();
        manager.setRealm(accountRealm);
        return manager;
    }

    /**
     * @Qualifier 同样通过它拿到 securityManager ，
     * 注入进 ShiroFilterFactoryBean 即可
     */
    @Bean
    public ShiroFilterFactoryBean shiroFilterFactoryBean(@Qualifier("securityManager") DefaultWebSecurityManager securityManager){
        ShiroFilterFactoryBean factoryBean = new ShiroFilterFactoryBean();
        factoryBean.setSecurityManager(securityManager);
        return factoryBean;
    }


}

@Bean
public ShiroFilterFactoryBean shiroFilterFactoryBean(@Qualifier("securityManager") DefaultWebSecurityManager securityManager){
    ShiroFilterFactoryBean factoryBean = new ShiroFilterFactoryBean();
    factoryBean.setSecurityManager(securityManager);

    //权限设置，限制请求的访问权限
    Map<String,String> map = new HashMap<>();               //规则存放在HashMap中
    map.put("/main","authc");                               
    //必须登录才能访问 main.html("/main"请求)
    map.put("/manager","perms[manager]");                   
    //当前用户必须拥有 manage 授权才能访问 manage.html("/manager"请求)
    map.put("/administrator","roles[administrator]");       
    //当前用户必须拥有 administrator 角色才能访问 administrator.html("/administrator"请求)
    factoryBean.setFilterChainDefinitionMap(map);           //把定义好的规则放进去
    
    //设置登录页面
    factoryBean.setLoginUrl("/login");
    
    return factoryBean;
}

这样就对访问这三个页面的请求进行了限制

这样再访问的话，就会默认自动跳转到 login.jsp 页面，但是我们没有这样的页面，

@PostMapping("/login")
public String login(String username, String password, Model model){
    //使用 Shiro 进行处理
    //拿到Subject： Shiro的一个抽象概念，包含了用户信息。
    Subject subject = SecurityUtils.getSubject();
    //把用户输入的信息传到token中封装
    UsernamePasswordToken token = new UsernamePasswordToken(username,password);
    /**
     * 执行了下面这一步，就会把登录请求交给Shiro去做，
     * 进入到 AccountRealm 中，对拿到的token，对比数据库，执行登录验证的操作，
     * 在 AccountRealm 中执行时，
     * 用户名错误，也就是根据用户名查不到数据库中对应的对象，会返回UnknownAccountException异常
     * 密码错误，返回IncorrectCredentialsException异常
     * 我们可以在这里捕获异常，并返回给登录页面错误信息提示
     */
    try {
        subject.login(token);
        return "index";
    } catch (UnknownAccountException e) {
        e.printStackTrace();
        model.addAttribute("msg","用户名不存在！");
        return "login";

    } catch (IncorrectCredentialsException e){
        model.addAttribute("msg","密码错误！");
        return "login";
    }

}

public class AccountRealm extends AuthorizingRealm {

    @Autowired

     * @param accountRealm  上一步注册到IOC中的对象，现在通过 @Qualifier 取出来并且注入到DefaultWebSecurityManager中
     * @Qualifier 通过对象的名字，从容器中取出需要的对象，名字默认是方法名
     * 同样要把它注入到IOC容器中，供最后干事的 ShiroFilterFactoryBean 使用
     * 所以下一步
     */
    @Bean
    public DefaultWebSecurityManager securityManager(@Qualifier("accountRealm") AccountRealm accountRealm){
        DefaultWebSecurityManager manager = new DefaultWebSecurityManager();
        manager.setRealm(accountRealm);
        return manager;
    }

    /**
     * @Qualifier 同样通过它拿到 securityManager ，
     * 注入进 ShiroFilterFactoryBean 即可
     */
    @Bean
    public ShiroFilterFactoryBean shiroFilterFactoryBean(@Qualifier("securityManager") DefaultWebSecurityManager securityManager){
        ShiroFilterFactoryBean factoryBean = new ShiroFilterFactoryBean();
        factoryBean.setSecurityManager(securityManager);

        //权限设置，限制请求的访问权限
        Map<String,String> map = new HashMap<>();               //规则存放在HashMap中
        map.put("/main","authc");                               //必须登录才能访问 main.html("/main"请求)
        map.put("/manage","perms[manage]");                     //当前用户必须拥有 manage 授权才能访问 manage.html("/manager"请求)
        map.put("/administrator","roles[administrator]");       //当前用户必须拥有 administrator 角色才能访问 administrator.html("/administrator"请求)
        factoryBean.setFilterChainDefinitionMap(map);           //把定义好的规则放进去
        //设置登录页面
        factoryBean.setLoginUrl("/login");

        //设置未授权页面，没权限访问的话，跳到这
        factoryBean.setUnauthorizedUrl("/unauth");

        return factoryBean;
    }

    @Bean
    public ShiroDialect shiroDialect(){
        return new ShiroDialect();
    }


}

package lover.alice.controller;

import lover.alice.entity.Account;
import org.apache.shiro.SecurityUtils;
import org.apache.shiro.authc.AuthenticationException;
import org.apache.shiro.authc.IncorrectCredentialsException;
import org.apache.shiro.authc.UnknownAccountException;
import org.apache.shiro.authc.UsernamePasswordToken;
import org.apache.shiro.subject.Subject;
import org.springframework.stereotype.Controller;
import org.springframework.ui.Model;
import org.springframework.web.bind.annotation.*;

@Controller
public class ShiroController {
    @GetMapping("{url}")                            //转到页面的处理
    public String tourl(@PathVariable String url){
        return url;
    }

    @PostMapping("/login")
    public String login(String username, String password, Model model){
        //使用 Shiro 进行处理
        //拿到Subject： Shiro的一个抽象概念，包含了用户信息。
        Subject subject = SecurityUtils.getSubject();
        //把用户输入的信息传到token中封装
        UsernamePasswordToken token = new UsernamePasswordToken(username,password);
        /**
         * 执行了下面这一步，就会把登录请求交给Shiro去做，
         * 进入到 AccountRealm 中，对拿到的token，对比数据库，执行登录验证的操作，
         * 在 AccountRealm 中执行时，
         * 用户名错误，也就是根据用户名查不到数据库中对应的对象，会返回UnknownAccountException异常
         * 密码错误，返回IncorrectCredentialsException异常
         * 我们可以在这里捕获异常，并返回给登录页面错误信息提示
         */
        try {
            subject.login(token);

            //拿到登录成功的用户的信息，存到Session中
            Account account = (Account) subject.getPrincipal();
            subject.getSession().setAttribute("account",account);

            return "index";
        } catch (UnknownAccountException e) {
            e.printStackTrace();
            model.addAttribute("msg","用户名不存在！");
            return "login";

        } catch (IncorrectCredentialsException e){
            model.addAttribute("msg","密码错误！");
            return "login";
        }

    }

    @GetMapping("/unauth")
    @ResponseBody
    public String unauth(){
        return "未授权，无法访问！";
    }


    @GetMapping("/logout")
    public String logout(){
        Subject subject = SecurityUtils.getSubject();
//        subject.getSession().removeAttribute("account");
        subject.logout();  //直接调这个方法 ，直接销毁Session
        return "login";
    }

}
